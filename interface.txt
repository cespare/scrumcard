 :user => <username>,
 :room => <room-name>,
 :votes => {
   <username> => <vote>
 }

  { :vote => <vote-value> }
POST /api/rooms/<roomname>/reset empty body; reset the votes
POST /api/rooms/<roomname>/login?redirect_to=<url>

GET /rooms/<roomname> serve up main room page; bounce to / if the username cookie is not set
